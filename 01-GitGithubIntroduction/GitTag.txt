git init - Yeni bir Git repository yaradır.
git clone <repository_url> - Mövcud bir repository-ni klonlayır.
git add <file(s)> - Yeni və ya dəyişilmiş faylları stagələyir.
git commit -m "commit message" - Yaddaşa yeni bir commit əlavə edir.
git status - Həyata keçirilməmiş dəyişiklikləri göstərir.
git log - Commit tarixçəsini göstərir.
git branch - Mövcud bölmələri göstərir.
git checkout <branch_name> - Başqa bir bölməyə keçir.
git merge <branch_name> - Bölmələri birləşdirir.
git pull - Mövcud yerli repository-dəki dəyişiklikləri yeniləyir.
git push - Yerli dəyişiklikləri uzaq repository-ə göndərir.
git remote add <name> <url> - Uzaq repository əlavə edir.
git remote -v - Uzaq repository-lərin siyahısını göstərir.
git fetch - Uzaq repository-də olan son dəyişiklikləri yükləyir.
git reset <file(s)> - Stagələnmiş faylları unstagələyir.
git reset --hard HEAD - Son commit-dən əvvəlki vəziyyətə qaytarır.
git checkout -b <new_branch_name> - Yeni bir bölmə yaradır və həmin bölməyə keçir.
git remote rm <name> - Uzaq repository-ni silir.
git diff - Dəyişiklikləri göstərir.
git tag <tag_name> - Yeni bir tag yaradır.
git stash - İşlədilməmiş dəyişiklikləri saklayır.
git stash apply - Son saklanan dəyişiklikləri geri alır.
git bisect - Git tarixçədə səhv tapmaq üçün istifadə olunur.
git rebase <branch_name> - Bölmənin tarixini başqa bir bölməyə əsaslanır.
git cherry-pick <commit_hash> - Yalnız bir commit-i əlavə edir.
git submodule - Submodule funksiyası.
git revert <commit_hash> - Bir commit-də olan dəyişiklikləri ləğv edir.
git remote show <name> - Uzaq repository haqqında ətraflı məlumat göstərir.
git clean - İşlənilən faylları və direktoriyaları silir.
git config - Git konfiqurasiyasını tənzimləyir.
